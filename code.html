function getValues() {
	let startValue = document.getElementById('startValue').value;
	let endValue = document.getElementById('endValue').value;

	let startNumber = parseInt(startValue);
	let endNumber = parseInt(endValue);

	let numberArray = generateNumbers(startNumber, endNumber);
	displayNumbers(numberArray);
}

	function generateNumbers(start, end) {

	let basketOfNumbers = [];

		basketOfNumbers.push(number);

	}

		function displayNumbers(numbers) { 
	
		let results = '';

		for (let index = 0; index < numbers.length; index=index + 1) { 

			let currentNumber=numbers[index]; 
			
			if (currentNumber % 2==0) { 
				results = results + "<tr><td class='evenNumber'>" + currentNumber + "" ;

				results = results + "<tr><td>" + currentNumber + "" ; } 
			
			} 
			
			let tableBody=document.getElementById('results');

Number Generator

This code uses 3 functions to display the results of 2 random numbers entered into the app with the even numbers displaying in bold red.

1. function getValues() receives the 2 numbers entered into app and parses both numbers for interger verification.

2. function generateNumbers(start, end) receives both numbers entered into app and runs a for loop starting a 0 and ending at 100. The for loop results are return and stored in basketOfNumbersarray and then get passed back to generateNumbers and stored in numberArray variable. On line 9, displayNumbers(numberArray) passes the previous for loop stored array numbers into function displayNumbers(numbers) located on line 28.

3. function displayNumbers(numbers) runs a for loop and uses a if/else statement to display all even numbers in bold red and all odd numbers in regular black text.
